### DDL — Script Completo (PostgreSQL)

```sql
-- =============================================
-- SISTEMA ARTE Y METALES — DDL COMPLETO
-- Motor: PostgreSQL 16
-- =============================================

-- 1. roles
CREATE TABLE roles (
    id BIGSERIAL PRIMARY KEY,
    nombre VARCHAR(80) NOT NULL UNIQUE,
    descripcion VARCHAR(200),
    activo BOOLEAN DEFAULT true,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);

-- 2. permisos
CREATE TABLE permisos (
    id BIGSERIAL PRIMARY KEY,
    nombre VARCHAR(120) NOT NULL,
    slug VARCHAR(120) NOT NULL UNIQUE,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);

-- 3. permiso_rol (PIVOT)
CREATE TABLE permiso_rol (
    id BIGSERIAL PRIMARY KEY,
    rol_id BIGINT NOT NULL REFERENCES roles(id) ON DELETE CASCADE,
    permiso_id BIGINT NOT NULL REFERENCES permisos(id) ON DELETE CASCADE,
    created_at TIMESTAMP,
    updated_at TIMESTAMP,
    UNIQUE(rol_id, permiso_id)
);

-- 4. users
CREATE TABLE users (
    id BIGSERIAL PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    email VARCHAR(255) NOT NULL UNIQUE,
    email_verified_at TIMESTAMP,
    password VARCHAR(255) NOT NULL,
    remember_token VARCHAR(100),
    rol_id BIGINT REFERENCES roles(id) ON DELETE SET NULL,
    activo BOOLEAN DEFAULT true,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);

-- 5. password_reset_tokens
CREATE TABLE password_reset_tokens (
    email VARCHAR(255) PRIMARY KEY,
    token VARCHAR(255) NOT NULL,
    created_at TIMESTAMP
);

-- 6. sessions
CREATE TABLE sessions (
    id VARCHAR(255) PRIMARY KEY,
    user_id BIGINT REFERENCES users(id) ON DELETE CASCADE,
    ip_address VARCHAR(45),
    user_agent TEXT,
    payload TEXT NOT NULL,
    last_activity INTEGER NOT NULL
);
CREATE INDEX idx_sessions_user_id ON sessions(user_id);
CREATE INDEX idx_sessions_last_activity ON sessions(last_activity);

-- 7. cache
CREATE TABLE cache (
    key VARCHAR(255) PRIMARY KEY,
    value TEXT NOT NULL,
    expiration INTEGER NOT NULL
);
CREATE INDEX idx_cache_expiration ON cache(expiration);

-- 8. cache_locks
CREATE TABLE cache_locks (
    key VARCHAR(255) PRIMARY KEY,
    owner VARCHAR(255) NOT NULL,
    expiration INTEGER NOT NULL
);
CREATE INDEX idx_cache_locks_expiration ON cache_locks(expiration);

-- 9. jobs
CREATE TABLE jobs (
    id BIGSERIAL PRIMARY KEY,
    queue VARCHAR(255) NOT NULL,
    payload TEXT NOT NULL,
    attempts SMALLINT NOT NULL,
    reserved_at INTEGER,
    available_at INTEGER NOT NULL,
    created_at INTEGER NOT NULL
);
CREATE INDEX idx_jobs_queue ON jobs(queue);

-- 10. job_batches
CREATE TABLE job_batches (
    id VARCHAR(255) PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    total_jobs INTEGER NOT NULL,
    pending_jobs INTEGER NOT NULL,
    failed_jobs INTEGER NOT NULL,
    failed_job_ids TEXT NOT NULL,
    options TEXT,
    cancelled_at INTEGER,
    created_at INTEGER NOT NULL,
    finished_at INTEGER
);

-- 11. failed_jobs
CREATE TABLE failed_jobs (
    id BIGSERIAL PRIMARY KEY,
    uuid VARCHAR(255) NOT NULL UNIQUE,
    connection TEXT NOT NULL,
    queue TEXT NOT NULL,
    payload TEXT NOT NULL,
    exception TEXT NOT NULL,
    failed_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 12. clientes
CREATE TABLE clientes (
    id BIGSERIAL PRIMARY KEY,
    nombre_completo VARCHAR(255) NOT NULL,
    telefono VARCHAR(20),
    correo VARCHAR(255),
    documento VARCHAR(25),
    direccion TEXT,
    observaciones TEXT,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);

-- 13. pedidos
CREATE TABLE pedidos (
    id BIGSERIAL PRIMARY KEY,
    codigo VARCHAR(30) NOT NULL UNIQUE,
    cliente_id BIGINT REFERENCES clientes(id) ON DELETE SET NULL,
    nombre_cliente VARCHAR(255) NOT NULL,
    telefono_cliente VARCHAR(20),
    documento_cliente VARCHAR(25),
    correo_cliente VARCHAR(255),
    tipo_producto VARCHAR(255) NOT NULL,
    tipo_entrega VARCHAR(20) DEFAULT 'local',
    direccion_entrega VARCHAR(255),
    referencia_entrega VARCHAR(255),
    distrito_entrega VARCHAR(120),
    codigo_postal_entrega VARCHAR(12),
    nombre_recibe VARCHAR(120),
    telefono_recibe VARCHAR(20),
    costo_delivery NUMERIC(10,2),
    detalle_trabajo TEXT,
    cantidad INTEGER DEFAULT 1,
    estado VARCHAR(30) DEFAULT 'registrado',
    estado_personalizacion VARCHAR(40) DEFAULT 'sin_iniciar',
    fecha_entrega_compromiso DATE,
    fecha_inicio_diseno DATE,
    fecha_aprobacion_diseno DATE,
    archivo_diseno_path VARCHAR(255),
    observaciones_personalizacion TEXT,
    monto_total NUMERIC(10,2),
    estado_pago VARCHAR(30) DEFAULT 'pendiente_adelanto',
    monto_adelanto NUMERIC(10,2),
    monto_saldo NUMERIC(10,2),
    observaciones TEXT,
    usuario_id BIGINT NOT NULL REFERENCES users(id),
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);

-- 14. pedido_diseno_archivos
CREATE TABLE pedido_diseno_archivos (
    id BIGSERIAL PRIMARY KEY,
    pedido_id BIGINT NOT NULL REFERENCES pedidos(id) ON DELETE CASCADE,
    archivo_path VARCHAR(255) NOT NULL,
    nombre_original VARCHAR(255) NOT NULL,
    mime_type VARCHAR(120),
    tamano_bytes BIGINT,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);

-- 15. pedido_orden_archivos
CREATE TABLE pedido_orden_archivos (
    id BIGSERIAL PRIMARY KEY,
    pedido_id BIGINT NOT NULL REFERENCES pedidos(id) ON DELETE CASCADE,
    archivo_path VARCHAR(255) NOT NULL,
    nombre_original VARCHAR(255),
    mime_type VARCHAR(120),
    tamano_bytes BIGINT,
    created_at TIMESTAMP,
    updated_at TIMESTAMP

-- 16. productos
CREATE TABLE productos (
    id BIGSERIAL PRIMARY KEY,

    nombre VARCHAR(255) NOT NULL,

CREATE TABLE producto_imagenes (

    archivo_path VARCHAR(255) NOT NULL,
    nombre_original VARCHAR(255) NOT NULL,
```
